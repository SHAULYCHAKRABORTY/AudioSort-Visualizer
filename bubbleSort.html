    Procedure BubbleSort(A: array of items)
                    n = length(A)  // Get the number of items in the array
                
                    // Repeat until no more swaps are needed
                    repeat
                        swapped = false  // Track if any swaps are made during this pass
                
                        // Traverse the array from start to end - 1
                        for i from 1 to n - 1 do
                            // Compare adjacent items
                            if A[i - 1] > A[i] then
                                // Swap if out of order
                                swap A[i - 1] with A[i]
                                swapped = true  // Indicate a swap was made
                
                        // Reduce the range for the next pass
                        n = n - 1
                
                    // Continue until no swaps are needed (array is sorted)
                    while swapped
