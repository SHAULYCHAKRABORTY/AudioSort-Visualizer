  • MERGE SORTING:

  • Algorithm:

    Procedure MergeSort(A, left, right)
                    if left < right then
                        mid = (left + right) / 2
                        MergeSort(A, left, mid)    // Recursively sort the first half
                        MergeSort(A, mid + 1, right) // Recursively sort the second half
                        Merge(A, left, mid, right)  // Merge the sorted halves
                
                Procedure Merge(A, left, mid, right)
                    n1 = mid - left + 1
                    n2 = right - mid
                    Create arrays L[1..n1+1] and R[1..n2+1]
                    
                    for i = 1 to n1 do
                        L[i] = A[left + i - 1]
                    for j = 1 to n2 do
                        R[j] = A[mid + j]
                
                    L[n1 + 1] = Infinity
                    R[n2 + 1] = Infinity
                
                    i = 1
                    j = 1
                    for k = left to right do
                        if L[i] <= R[j] then
                            A[k] = L[i]
                            i = i + 1
                        else
                            A[k] = R[j]
                            j = j + 1
