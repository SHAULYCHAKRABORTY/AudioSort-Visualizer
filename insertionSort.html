    Procedure InsertionSort(A: array of items)
                    // Start from the second element (index 1) since a single element is already sorted
                    for i from 1 to length(A) - 1 do
                        // Current element to be compared and inserted
                        key = A[i]
                        
                        // Index of the previous element
                        j = i - 1
                        
                        // Move elements of A[0..i-1] that are greater than key to one position ahead
                        while j >= 0 and A[j] > key do
                            A[j + 1] = A[j]
                            j = j - 1
                        
                        // Insert the key into its correct position
                        A[j + 1] = key
